/* Calcula o gcd de a e b em complexidade O(log(min(a, b)))  */
int gcd(int a, int b){
    int d;
    int r;

    if(b == 0)
        return a;

    d = a/b;
    r = a - d*b;

    return gcd(b, r);
}

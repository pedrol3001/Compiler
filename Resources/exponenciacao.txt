/* Calcula a^b em complexidade O(log(b)) */
int pow(int a, int b){
    int res;
    int d;
    int r;

    if(b == 0)
        return 1;
    
    res = pow(a, b/2);

    d = a/2;
    r = a - d*2;

    if(r)
        return res*res*a;
    else
        return res*res;
}

int main(void){
    int a;
    int b;
    
    input a;
    input b;

    output pow(a, b);
}
